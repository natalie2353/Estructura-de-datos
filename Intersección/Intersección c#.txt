using System;
using System.Linq;

class Program {
    static void Main() {
        string[] grupo1 = {"Delfín", "Tiburón", "Ballena", "Pulpo"};
        string[] grupo2 = {"Pulpo", "Medusa", "Tiburón", "Estrella de mar"};

        var interseccion = grupo1.Intersect(grupo2);

        foreach(var a in interseccion)
            Console.WriteLine(a);
    }
}
